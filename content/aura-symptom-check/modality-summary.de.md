<script type="module">
import { getDataFromLocalStorage } from '/js/aura-symptom-check/dataRetrieval.js';
import { generateNarrativeSummary } from '/js/aura-symptom-check/summaryGeneration.js';
import { t } from '/js/decision-flow/i18n.js';
import { registerRoboto } from '/js/decision-flow/fonts/roboto.js';

const FONT = "Roboto";

async function loadImageAsBase64(url) {
  return new Promise((resolve, reject) => {
    const img = new Image();
    img.crossOrigin = "anonymous";
    img.onload = () => {
      const canvas = document.createElement("canvas");
      const context = canvas.getContext("2d");
      canvas.width = img.width;
      canvas.height = img.height;
      context.drawImage(img, 0, 0, img.width, img.height);
      resolve({ base64: canvas.toDataURL("image/png"), width: img.width, height: img.height });
    };
    img.onerror = reject;
    img.src = url;
  });
}

async function addPdfHeader(pdf, pageWidth) {
  const marginLeft = 20;
  const titleY = 15;
  const disclaimerY = 21;
  const lineY = 23;
  const logoMarginTop = 6;

  pdf.setFont(FONT, "normal").setFontSize(14);
  pdf.text(t("flow.report.title", "Migraine Aura Report"), marginLeft, titleY);

  try {
    const { base64, width, height } = await loadImageAsBase64("/images/report-logo.png");
    const logoHeight = 10;
    const logoWidth = (logoHeight * width) / height;
    pdf.addImage(base64, "PNG", pageWidth - logoWidth - marginLeft, logoMarginTop, logoWidth, logoHeight);
  } catch (e) {
    console.warn("Logo missing:", e);
  }

  pdf.setFontSize(9).setFont(FONT, "normal");
  pdf.setTextColor(0);
  const lead = t("flow.report.headerDisclaimer.lead", "This automated report adheres to ICHD-3 criteria and is");
  const emph = t("flow.report.headerDisclaimer.emph", " not a substitute ");
  const tail = t("flow.report.headerDisclaimer.tail", "for professional medical diagnosis.");
  pdf.text(lead, marginLeft, disclaimerY);
  pdf.setTextColor(200, 0, 0);
  pdf.text(emph, marginLeft + pdf.getTextWidth(lead), disclaimerY);
  pdf.setTextColor(0);
  pdf.text(tail, marginLeft + pdf.getTextWidth(lead + emph), disclaimerY);
  pdf.line(marginLeft, lineY, pageWidth - marginLeft, lineY);
}

function addPdfFooter(pdf, pageNum, pageHeight, pageWidth) {
  const marginLeft = 20;
  const marginBottom = 20;

  pdf.setFontSize(8).setTextColor(120).setFont(FONT, "normal");
  pdf.text(t("flow.report.footer.generatedBy", "Generated by Migraine Aura Foundation"), marginLeft, pageHeight - marginBottom);
  pdf.text(t("flow.report.footer.url", "www.migraine-aura-foundation.org"), pageWidth - marginLeft, pageHeight - marginBottom, { align: "right" });
  pdf.setFontSize(7).setTextColor(150);
  pdf.text(t("flow.report.footer.disclaimerNote", "This assessment is not a diagnosis. See full disclaimer on last page."), marginLeft, pageHeight - marginBottom + 5);
}

document.addEventListener("DOMContentLoaded", () => {
  const output = document.getElementById("summaryOutput");
  const downloadBtn = document.getElementById("downloadPdfBtn");
  downloadBtn.textContent = t("flow.report.button.download", "Download PDF Report");

  const data = getDataFromLocalStorage();
  output.innerHTML = generateNarrativeSummary(data);

  downloadBtn.addEventListener("click", async () => {
    if (!window.jspdf) {
      alert(t("flow.report.alert.jspdfFailed", "jsPDF failed to load. Please try again later."));
      return;
    }

    const { jsPDF } = window.jspdf;
    const pdf = new jsPDF({ orientation: "portrait", unit: "mm", format: "a4" });
    await registerRoboto(pdf);
    pdf.setFont(FONT, "normal");

    const pageWidth = pdf.internal.pageSize.getWidth();
    const pageHeight = pdf.internal.pageSize.getHeight();
    const marginTop = 35;
    const marginBottom = 20;
    const marginLeft = 20;
    const marginRight = 20;
    const usableWidth = pageWidth - marginLeft - marginRight;

    let y = marginTop;
    await addPdfHeader(pdf, pageWidth);

    const paragraphs = output.innerText.split("\n");
    const lineHeight = 6;

    for (const text of paragraphs) {
      if (y + lineHeight > pageHeight - marginBottom) {
        pdf.addPage();
        await addPdfHeader(pdf, pageWidth);
        y = marginTop;
      }
      const lines = pdf.splitTextToSize(text, usableWidth);
      lines.forEach((line) => {
        pdf.setFont(FONT, "normal").setFontSize(10).setTextColor(0);
        pdf.text(line, marginLeft, y);
        y += lineHeight;
      });
    }

    // Disclaimer page — the footer note on every page promises "see full
    // disclaimer on last page", so render it here as the final page.
    pdf.addPage();
    await addPdfHeader(pdf, pageWidth);
    pdf.setFont(FONT, "normal").setFontSize(13).setTextColor(0);
    pdf.text(t("flow.report.disclaimer.heading", "⚠️ Disclaimer"), marginLeft, marginTop);
    pdf.setFontSize(10);
    const disclaimerBody = t(
      "flow.report.disclaimer.body",
      "This assessment was conducted using a self-report tool developed by the Migraine Aura Foundation. It is not a medical diagnosis.\n\nNeurological symptoms can have many causes. This assumes your symptoms are not better explained by another medical condition such as epilepsy, stroke, multiple sclerosis, or eye disorders.\n\nIf your symptoms are new, worsening, or unusual, seek medical advice immediately.",
    );
    pdf.text(pdf.splitTextToSize(disclaimerBody, usableWidth), marginLeft, marginTop + 10);

    addPdfFooter(pdf, 2, pageHeight, pageWidth);
    pdf.save(t("flow.report.fileName", "Migraine-Aura-Report.pdf"));
    localStorage.removeItem("redirectToDemographicInfo");
  });
});
